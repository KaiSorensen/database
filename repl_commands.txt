create_object(demo_people);
create_attribute(demo_people, name, string);
create_attribute(demo_people, age, int);
create_attribute(demo_people, bonus, int);
create_object(demo_employees, extends(demo_people));
read_object(demo_employees, include_attributes(), include_row_count());
create_rows(demo_people, values(row(name("Ada"), age(20), bonus(5)), row(name("Bob"), age(30), bonus(7)), row(name("Cara"), age(25), bonus(9))));
read(demo_people, attributes(name, age, bonus));
read(demo_people, attributes(name, age), where(greater_than(age, 20)));
read(demo_people, derive(total, add(age, bonus)));
read(demo_people, summarize(max_age, max(age)));
create_attribute(demo_people, total, int, derive(add(age, bonus)));
read_attribute(demo_people, total, include_type());
update(demo_people, set(age, add(age, bonus)), where(greater_than(age, 20)));
read(demo_people, attributes(name, age, bonus, total));
update(demo_people, set(bonus, max(age)), where(greater_than(age, 20)));
read(demo_people, attributes(name, age, bonus, total));
create_object(demo_pets);
create_attribute(demo_pets, owner_name, string);
create_attribute(demo_pets, pet_name, string);
create_rows(demo_pets, values(row(owner_name("Ada"), pet_name("Milo")), row(owner_name("Cara"), pet_name("Nora")), row(owner_name("Zed"), pet_name("Otis"))));
read(join(demo_people, demo_pets, on(equals(demo_people.name, demo_pets.owner_name))), attributes(demo_people.name, demo_pets.pet_name));
delete_rows(demo_people, where(greater_than(age, 30)));
delete_attributes(demo_people, attributes(total));
update_attribute(demo_people, bonus, rename(extra_bonus));
read_object(demo_people, include_attributes(), include_row_count());
